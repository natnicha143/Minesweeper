import sys
from GameModel import Grid
from GameView import GameView
from TopMenu import TopMenu
from GameController import GameController
from TopMenu import TopMenu
from tkinter import *

class MineSweeper:
    def __init__(self):
        #create main window
        self.root = Tk()
        self.root.title("Mine Sweeper")
        self.root["bg"] = "black"
        self.root.resizable(width=False, height=False)
        self.model = Grid()
        self.controller = BoardController(self.model)
        self.view = GameView(self.controller, self.model, self.width, self.height)
        self.topmenu = TopMenu(self.root, self.model, self.view, self.board)
        self.root.mainloop()

MineSweeper()

class GameController:
    def __init__(self, model, board):
        self.model = model
        self.board = board
        self.neighbours = self.model.get_neighbours()

class Grid:
    def __init__(self):
        self.height = 10
        self.width = 15
        self.mines = 20
        self.mines_remain = self.mines
        self.tiles_revealed = 0
        self.init_time = time.time()
        self.neighbours = self.populate_game()
        self.backing_grid = self.get_grid()
